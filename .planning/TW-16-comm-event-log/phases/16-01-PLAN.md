# 16-01 — Event Log Store & Types

## Objective
Create the Pinia store and TypeScript types for a categorized, timestamped event log. Replace the single `message` ref in the ship store.

## Context
- See `../BRIEF.md` and `../ROADMAP.md`
- Currently `ship.message` is a single string ref — every action overwrites the previous message
- No history, no categorization, no timestamps
- NavigationView only tracks sector visits, not trades/combat/upgrades/events

### 2. Create event log store
Create `web/game/src/stores/eventLog.ts`:

```ts
export const useEventLogStore = defineStore('eventLog', () => {
  const events = ref<GameEvent[]>([]);
  let nextId = 0;

  function addEvent(tag: EventTag, text: string, sector?: number) {
    events.value.push({
      id: nextId++,
      tag,
      text,
      sector,
      timestamp: Date.now(),
    });
    // Keep last 100 entries
    if (events.value.length > 100) {
      events.value = events.value.slice(-100);
    }
  }

  function clear() {
    events.value = [];
    nextId = 0;
  }

  const recentEvents = computed(() => events.value.slice(-20));
  const lastEvent = computed(() => events.value[events.value.length - 1] ?? null);

  return { events, recentEvents, lastEvent, addEvent, clear };
});
```
